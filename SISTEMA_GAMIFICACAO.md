#### **Tabela: user_points**
```sql
CREATE TABLE user_points (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    points INT DEFAULT 0,
    level VARCHAR(50) DEFAULT 'Semente de Alecrim',
    level_number INT DEFAULT 1,
    next_level_points INT DEFAULT 100,
    total_earned INT DEFAULT 0,
    created_at TIMESTAMP,
    updated_at TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
```

#### **Tabela: point_transactions**
```sql
CREATE TABLE point_transactions (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    points INT NOT NULL,
    action VARCHAR(100) NOT NULL,
    description TEXT,
    reference_type VARCHAR(50),
    reference_id BIGINT,
    created_at TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
```

#### **Tabela: user_badges**
```sql
CREATE TABLE user_badges (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    badge_code VARCHAR(50) NOT NULL,
    badge_name VARCHAR(100) NOT NULL,
    badge_description TEXT,
    earned_at TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    UNIQUE KEY (user_id, badge_code)
);
```

### **2. Backend (Laravel)**

#### **Model: UserPoint.php**
```php
<?php

namespace App\Modules\Gamification\Models;

use Illuminate\Database\Eloquent\Model;
use App\Modules\Auth\Models\User;

class UserPoint extends Model
{
    protected $fillable = [
        'user_id',
        'points',
        'level',
        'level_number',
        'next_level_points',
        'total_earned',
    ];

    protected $casts = [
        'points' => 'integer',
        'level_number' => 'integer',
        'total_earned' => 'integer',
    ];

    public function user()
    {
        return $this->belongsTo(User::class);
    }

    public function transactions()
    {
        return $this->hasMany(PointTransaction::class, 'user_id', 'user_id');
    }

    // Níveis definidos
    public static $levels = [
        1 => ['name' => 'Semente de Alecrim', 'min' => 0, 'max' => 99, 'discount' => 0],
        2 => ['name' => 'Broto de Camomila', 'min' => 100, 'max' => 299, 'discount' => 5],
        3 => ['name' => 'Folha de Hortelã', 'min' => 300, 'max' => 599, 'discount' => 10],
        4 => ['name' => 'Flor de Lavanda', 'min' => 600, 'max' => 999, 'discount' => 15],
        5 => ['name' => 'Raiz de Gengibre', 'min' => 1000, 'max' => 1999, 'discount' => 20],
        6 => ['name' => 'Árvore Sagrada', 'min' => 2000, 'max' => 4999, 'discount' => 25],
        7 => ['name' => 'Guardião da Floresta', 'min' => 5000, 'max' => 999999, 'discount' => 30],
    ];

    // Calcular nível baseado em pontos
    public function calculateLevel()
    {
        foreach (self::$levels as $num => $level) {
            if ($this->points >= $level['min'] && $this->points <= $level['max']) {
                $this->level_number = $num;
                $this->level = $level['name'];
                
                // Próximo nível
                if (isset(self::$levels[$num + 1])) {
                    $this->next_level_points = self::$levels[$num + 1]['min'];
                } else {
                    $this->next_level_points = null; // Max level
                }
                
                $this->save();
                return;
            }
        }
    }

    // Adicionar pontos
    public function addPoints(int $points, string $action, string $description = null)
    {
        $this->points += $points;
        $this->total_earned += $points;
        $this->calculateLevel();
        
        // Registrar transação
        PointTransaction::create([
            'user_id' => $this->user_id,
            'points' => $points,
            'action' => $action,
            'description' => $description,
        ]);
        
        return $this;
    }

    // Obter desconto do nível atual
    public function getDiscount(): int
    {
        return self::$levels[$this->level_number]['discount'] ?? 0;
    }
}
```

#### **Service: GamificationService.php**
```php
<?php

namespace App\Modules\Gamification\Services;

use App\Modules\Auth\Models\User;
use App\Modules\Gamification\Models\UserPoint;
use App\Modules\Gamification\Models\UserBadge;

class GamificationService
{
    // Inicializar pontos para novo usuário
    public function initializeUser(User $user)
    {
        return UserPoint::create([
            'user_id' => $user->id,
            'points' => 20, // Bônus de cadastro
            'level' => 'Semente de Alecrim',
            'level_number' => 1,
            'next_level_points' => 100,
            'total_earned' => 20,
        ]);
    }

    // Recompensar por compra
    public function rewardPurchase(User $user, float $orderTotal)
    {
        $userPoint = UserPoint::where('user_id', $user->id)->first();
        
        // 10 pontos por cada R$ 10
        $points = (int)floor($orderTotal / 10) * 10;
        
        // Bônus primeira compra
        $isFirstPurchase = $user->orders()->count() === 1;
        if ($isFirstPurchase) {
            $points += 50;
            $this->awardBadge($user, 'first_purchase', 'Primeira Compra', 'Ritual Iniciado');
        }
        
        $userPoint->addPoints($points, 'purchase', "Compra de R$ {$orderTotal}");
        
        return $userPoint;
    }

    // Recompensar por avaliação
    public function rewardReview(User $user)
    {
        $userPoint = UserPoint::where('user_id', $user->id)->first();
        $userPoint->addPoints(30, 'review', 'Avaliação de produto');
        
        // Badge para 20 avaliações
        if ($user->reviews()->count() === 20) {
            $this->awardBadge($user, 'communicator', 'Comunicador', '20 avaliações realizadas');
        }
        
        return $userPoint;
    }

    // Recompensar por indicação
    public function rewardReferral(User $referrer, User $referred)
    {
        $userPoint = UserPoint::where('user_id', $referrer->id)->first();
        $userPoint->addPoints(100, 'referral', "Indicou {$referred->name}");
        
        // Badge para 10 indicações
        $referralCount = $referred->where('referred_by', $referrer->id)->count();
        if ($referralCount === 10) {
            $this->awardBadge($referrer, 'influencer', 'Influenciador', '10 indicações');
        }
        
        return $userPoint;
    }

    // Conceder badge
    public function awardBadge(User $user, string $code, string $name, string $description)
    {
        return UserBadge::firstOrCreate(
            ['user_id' => $user->id, 'badge_code' => $code],
            [
                'badge_name' => $name,
                'badge_description' => $description,
            ]
        );
    }
}
```

### **3. Frontend (React/Next.js)**

#### **Componente: LevelProgress.tsx**
```tsx
"use client"

import { Card, CardContent } from "@/components/ui/card"
import { Badge } from "@/components/ui/badge"
import { Progress } from "@/components/ui/progress"

interface LevelProgressProps {
  level: string
  levelNumber: number
  currentPoints: number
  nextLevelPoints: number
  discount: number
}

export function LevelProgress({ 
  level, 
  levelNumber, 
  currentPoints, 
  nextLevelPoints,
  discount 
}: LevelProgressProps) {
  const progress = nextLevelPoints 
    ? (currentPoints / nextLevelPoints) * 100 
    : 100

  const levelEmojis = {
    1: "🌱",
    2: "🌿",
    3: "🍃",
    4: "💜",
    5: "🌾",
    6: "🌳",
    7: "👑"
  }

  return (
    <Card className="border-none shadow-lg bg-gradient-to-br from-sage to-sage/80 text-white">
      <CardContent className="p-6 space-y-4">
        <div className="flex items-center justify-between">
          <div>
            <p className="text-white/80 text-sm font-medium uppercase tracking-widest">
              Nível de Membro
            </p>
            <h3 className="text-2xl font-bold font-heading flex items-center gap-2">
              {levelEmojis[levelNumber]} {level}
            </h3>
          </div>
          {discount > 0 && (
            <Badge variant="dourado" size="lg">
              {discount}% OFF
            </Badge>
          )}
        </div>

        {nextLevelPoints && (
          <div className="space-y-2">
            <div className="flex justify-between text-sm">
              <span>{currentPoints} pts</span>
              <span>{nextLevelPoints} pts</span>
            </div>
            <Progress value={progress} className="h-2" />
            <p className="text-xs text-white/70">
              Faltam {nextLevelPoints - currentPoints} pontos para o próximo nível
            </p>
          </div>
        )}

        {levelNumber === 7 && (
          <p className="text-xs text-white/70">
            🎉 Você alcançou o nível máximo!
          </p>
        )}
      </CardContent>
    </Card>
  )
}
```
